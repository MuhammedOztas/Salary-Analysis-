# Gini Katsayısı Hesaplama
def calculate_gini(salaries):
    sorted_salaries = np.sort(salaries)
    n = len(salaries)
    cumsum = np.cumsum(sorted_salaries)
    return (n + 1 - 2 * np.sum(cumsum) / cumsum[-1]) / n

def convert_to_usd(amount, currency, rates):
    if currency == 'TRY':
        return amount / rates['TRY_USD']
    elif currency == 'EUR':
        return amount * rates['EUR_USD']
    return amount
